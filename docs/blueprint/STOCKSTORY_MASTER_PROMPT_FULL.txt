// Scoring Algorithm
export function calculateScore(stock: Stock, preset: ScannerPreset): number {
  const factors = {
    qualityScore: calculateQuality(stock), // ROE, ROIC, margin quality
    valuationScore: calculateValuation(stock), // PE, PB, EV/EBITDA, DCF
    growthScore: calculateGrowth(stock), // Revenue & profit growth
    momentumScore: calculateMomentum(stock), // RSI, MACD, trend
    stabilityScore: calculateStability(stock), // Debt, beta, VIX
  };

  const weightedScore = 
    (factors.qualityScore * preset.weights.quality) +
    (factors.valuationScore * preset.weights.valuation) +
    (factors.growthScore * preset.weights.growth) +
    (factors.momentumScore * preset.weights.momentum) +
    (factors.stabilityScore * preset.weights.stability);

  return Math.round(weightedScore);
}

    Focus on: quality, growth, valuation, or momentum—whichever is most relevant.
    Be specific and avoid generic language.
  `;

    Promise.all([
      fetchStock(symbol),
      fetchChart(symbol, timeframe),
      fetchFinancials(symbol),
      fetchNews(symbol),
    ]).then(([s, c, f, n]) => {
      setStock(s);
      setChart(c);
      setFinancials(f);
      setNews(n);
      setLoading(false);
    });
  }, [symbol, timeframe]);

  // Keyboard shortcuts
  useEffect(() => {
    const handleKeyPress = (e: KeyboardEvent) => {
      if (e.key === 't') trackStock(symbol);
      if (e.key === 'c') navigateToCompare(symbol);
      if (e.key === '?') showKeyboardShortcuts();
    };
    window.addEventListener('keydown', handleKeyPress);
    return () => window.removeEventListener('keydown', handleKeyPress);
  }, [symbol]);

  if (loading) return <LoadingState />;

  return (
    <div style={styles.container}>
      {/* STICKY HEADER */}
      <Header stock={stock} />

      {/* CHART */}
      <ChartSection
        stock={stock}
        chart={chart}
        timeframe={timeframe}
        onTimeframeChange={setTimeframe}
      />

      {/* HEALTHOMETER + CONVICTION */}
      <HealthometerSection stock={stock} />

      {/* METRICS GRID */}
      <MetricsGrid metrics={stock.metrics} />

      {/* COMPANY DETAILS */}
      <CompanyDetailsCard stock={stock} />

      {/* FINANCIAL HISTOGRAMS */}
      <FinancialHistograms data={financials} />

      {/* BULL / BEAR CASE */}
      <ThesisSection stock={stock} />

      {/* WHAT CHANGED */}
      <WhatChangedSection stock={stock} />

      {/* WHAT TO WATCH */}
      <WhatToWatchSection stock={stock} />

      {/* NEWS FEED */}
      <NewsSection news={news} symbol={symbol} />

      {/* ACTION BUTTONS */}
      <ActionButtons stock={stock} />
    </div>
  );
}

// Smooth number counter animation
export function AnimatedNumber({ value, decimals = 0 }: Props) {
  const [displayValue, setDisplayValue] = useState(0);
  const animationRef = useRef<number>();

  useEffect(() => {
    const startValue = displayValue;
    const difference = value - startValue;
    let frame = 0;
    const totalFrames = 30; // 0.5s at 60fps

    const animate = () => {
      frame++;
      const progress = frame / totalFrames;
      const easeOut = 1 - Math.pow(1 - progress, 3);
      const newValue = startValue + difference * easeOut;

      setDisplayValue(parseFloat(newValue.toFixed(decimals)));

      if (frame < totalFrames) {
        animationRef.current = requestAnimationFrame(animate);
      }
    };

    animationRef.current = requestAnimationFrame(animate);
    return () => cancelAnimationFrame(animationRef.current!);
  }, [value, decimals]);

  return <span>{displayValue}</span>;
}

  'Escape': { action: 'close', description: 'Close dialog / Go back' },

// Hook for handling shortcuts
export function useGlobalKeyboardShortcuts() {
  const navigate = useNavigate();
  const [selectedIndex, setSelectedIndex] = useState(0);

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      const key = `${e.ctrlKey || e.metaKey ? 'cmd+' : ''}${
        e.shiftKey ? 'shift+' : ''
      }${e.key.toLowerCase()}`;

      if (GLOBAL_SHORTCUTS[key]) {
        e.preventDefault();
        const { action } = GLOBAL_SHORTCUTS[key];
        handleShortcutAction(action);
      }

      // List navigation
      if (e.key === 'ArrowUp') {
        e.preventDefault();
        setSelectedIndex(prev => Math.max(0, prev - 1));
      }
      if (e.key === 'ArrowDown') {
        e.preventDefault();
        setSelectedIndex(prev => prev + 1);
      }
      if (e.key === 'Enter') {
        e.preventDefault();
        // Select current item
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, []);

  return { selectedIndex };
}

const handleShortcutAction = (action: string) => {
  switch (action) {
    case 'openSearch':
      // Open command palette
      break;
    case 'newResearch':
      // Start new research
      break;
    // ...etc
  }
};
